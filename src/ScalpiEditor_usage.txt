This is ScalpiEditor - "heirarhy" text editor.

for open existing file just write file name as argument:
$ ScalpiEditor "~/.bashrc"

main key bindings   {
  exit                   [Ctrl]  +  [q]  or  [F10]
  save                   [Ctrl]  +  [s]
  move cursour:          just use arrows
}
folding             {
  parse with brackets    [Ctrl]  +  [e]
  parse with indent      [Ctrl]  +  [r]
  unparse                [Ctrl]  +  [u]
  enter block            [Tab]
  leave block            [Esc] 
}
navigation          {
  go to start of line    [Home]
  go to end   of line    [End]
  go to first line       [Ctrl]  +  [Up]
  go to last  line       [Ctrl]  +  [Down]
  go to line from number [Ctrl]  +  [g]
  go to left   word      [Ctrl]  +  [Left]
  go to right  word      [Ctrl]  +  [Right]
}
edit                {
  insert char            [any]
  new line / divide line [Enter]
  delete prev char       [BackSpace]
  delete next char       [Del]
  clear line             [Shift] +  [Del]
  duplicate              [Ctrl]  +  [d]
  cut                    [Ctrl]  +  [x]
  copy (external)        [Ctrl]  +  [c]
  paste                  [Ctrl]  +  [v]
  paste (external)       [Alt]   +  [v]
  move line down         [Alt]   +  [Down]
  move line up           [Alt]   +  [Up]
}
mark                {
  mark                   [Alt]  +  [m]
  jump to mark           [Alt]  +  [j]
}
indent              {
  fix indent:    [Ctrl]  +  [p]
  remove indent: [alt]   +  [p]
}
