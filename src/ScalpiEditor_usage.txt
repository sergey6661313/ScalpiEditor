This is ScalpiEditor - "heirarhy" text editor.

for open existing file just write file name as argument:
$ ScalpiEditor "~/.bashrc"

key bindings:
exit                   [Ctrl] + [q]
save                   [Ctrl] + [s]
up                     [Up]
down                   [Down]
enter block            [Ctrl] + [i]
leave block            [Ctrl] + [o]


navigation {
left                   [Left]
right                  [Right]
go to start  line      [Home]
go to end    line      [End]
go to root             [Ctrl]  + [Slash]
}

edit {
insert char            [any]
delete prev char       [BackSpace]
delete next char       [Del]
cut                    [Ctrl]  + [x]
insert line upper      [Shift] + [Enter]
insert line below      [Alt]   + [Enter]
clear line             [Ctrl]  + [c]
swap with upper  line  [Ctrl]  + [j]
swap with bottom line  [Ctrl]  + [k]
}

unrealized {
go to left   word      [Ctrl]  + [Left]
go to right  word      [Ctrl]  + [Right]
set block selected     [Ctrl]  + [a]
select left  symbol    [Shift] + [←]
select right symbol    [Shift] + [→]
create new mark        [Ctrl]  + [slash]
jump to prev mark      [Ctrl]  + [↑]
jump to next mark      [Ctrl]  + [↓]
swap to commentary     [Alt]   + [Slash]
go to line from number [Ctrl]  + [g]
}
