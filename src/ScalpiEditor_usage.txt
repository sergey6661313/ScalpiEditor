This is ScalpiEditor - "heirarhy" text editor.

for open existing file just write file name as argument:
$ ScalpiEditor "~/.bashrc"

main key bindings {
  exit                   [Ctrl]  +  [q]
  save                   [Ctrl]  +  [s]
  move cursour:          just use arrows
}
folding     {
  parse with brackets    [Ctrl]  +  [e]
  parse with indent      [Ctrl]  +  [r]
  unparse                [Ctrl]  +  [u]
  enter block            [Tab]
  leave block            [Esc] 
}
navigation  {
  go to start of line    [Home]
  go to end   of line    [End]
  go to last line        [Ctrl]  +  [l]
  go to line from number [Ctrl]  +  [g]
}
edit        {
  insert char            [any]
  new line / divide line [Enter]
  delete prev char       [BackSpace]
  delete next char       [Del]
  clear line             [Ctrl]  +  [BackSpace]
  duplicate              [Ctrl]  +  [d]
  cut                    [Ctrl]  +  [x]
  copy (external)        [Ctrl]  +  [c]
  paste                  [Ctrl]  +  [v]
  move line down         [Ctrl]  +  [j]
  move line up           [Ctrl]  +  [k]
}

unrealized  {
  select left  symbol    [Shift] +  [Left]
  select right symbol    [Shift] +  [Right]
  go to left   word      [Ctrl]  +  [Left]
  go to right  word      [Ctrl]  +  [Right]
  create new mark        [Ctrl]  +  [slash]
  jump to prev mark      [Ctrl]  +  [Up]
  jump to next mark      [Ctrl]  +  [Down]
  swap to commentary     [Alt]   +  [Slash]
}
